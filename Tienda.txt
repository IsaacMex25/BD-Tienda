/*Procedimiento para insertar productos*/
DELIMITER //
DROP PROCEDURE if EXISTS p_insertarproductos;
CREATE PROCEDURE p_insertarproductos 
(
 IN _nombre VARCHAR(50),
 IN _descripcion VARCHAR(255),
 IN _precio decimal(6,2)
)
BEGIN
 INSERT INTO tienda (nombre, descripcion, precio) VALUES 
 (_nombre, _descripcion, _precio);
END //
DELIMITER ;
